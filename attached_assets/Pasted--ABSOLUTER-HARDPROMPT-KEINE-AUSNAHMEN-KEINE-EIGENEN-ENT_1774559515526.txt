## HOOK: useReveal.ts — EXAKT SO

```typescript
import { useEffect, useRef } from "react";

export function useReveal() {
  const ref = useRef<HTMLDivElement>(null);
  useEffect(() => {
    const el = ref.current;
    if (!el) return;
    const observer = new IntersectionObserver(
      (entries) => entries.forEach((e) => {
        if (e.isIntersecting) e.target.classList.add("visible");
      }),
      { threshold: 0.1 }
    );
    el.querySelectorAll(".reveal").forEach((t) => observer.observe(t));
    return () => observer.disconnect();
  }, []);
  return ref;
}
```

## API-ROUTE — EXAKT SO

### server/routes.ts — POST /api/interesse
```typescript
import { z } from 'zod';
import { db } from '../lib/db';
import { interessenten } from '../lib/schema';

const interesSchema = z.object({
  name: z.string().min(1).max(100),
  personen: z.string(),
  statement: z.string().max(500).optional(),
  song: z.string().max(200).optional(),
});

// POST /api/interesse
export async function handleInteresse(req: Request): Promise<Response> {
  const body = await req.json();
  const parsed = interesSchema.safeParse(body);
  if (!parsed.success) {
    return Response.json({ error: 'Ungültige Eingabe' }, { status: 400 });
  }
  await db.insert(interessenten).values(parsed.data);
  return Response.json({ success: true });
}
```
