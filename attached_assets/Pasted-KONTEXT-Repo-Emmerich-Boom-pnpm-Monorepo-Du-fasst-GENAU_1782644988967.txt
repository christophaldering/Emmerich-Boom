Imports die du brauchst (oben in der Datei ergänzen):
  import { db, scanLog, anmeldungTicketsTable, thekeProfileTable } from "@workspace/db";
  import { desc, eq, isNotNull, gte } from "drizzle-orm";

Logik des Endpunkts:

A) NEUANKÖMMLINGE — letzte 90 Sekunden aus scan_log:
   SELECT scan_log.id, scan_log.person_name, scan_log.scanned_at,
          theke_profile.lauter_song
   FROM scan_log
   LEFT JOIN anmeldung_tickets ON anmeldung_tickets.ticket_code = scan_log.ticket_code
   LEFT JOIN theke_profile ON theke_profile.anmeldung_ticket_id = anmeldung_tickets.id
   WHERE scan_log.result = 'ok'
     AND scan_log.scanned_at >= NOW() - INTERVAL '90 seconds'
   ORDER BY scan_log.scanned_at DESC
   LIMIT 10

   In Drizzle:
   const cutoff = new Date(Date.now() - 90_000);
   const neuankoemmlinge = await db
     .select({
       id:          scanLog.id,
       person_name: scanLog.person_name,
       scanned_at:  scanLog.scanned_at,
       lauter_song: thekeProfileTable.lauter_song,
     })
     .from(scanLog)
     .leftJoin(anmeldungTicketsTable, eq(anmeldungTicketsTable.ticket_code, scanLog.ticket_code))
     .leftJoin(thekeProfileTable, eq(thekeProfileTable.anmeldung_ticket_id, anmeldungTicketsTable.id))
     .where(and(eq(scanLog.result, "ok"), gte(scanLog.scanned_at, cutoff)))
     .orderBy(desc(scanLog.scanned_at))
     .limit(10);

   Hinweis: `and` aus drizzle-orm importieren.

B) ANWESENDE — alle erfolgreich eingescannten Gäste heute (eingelassen_am NOT NULL),
   mit tafel_ok = true aus theke_profile:
   const anwesende = await db
     .select({
       person_name:  anmeldungTicketsTable.person_name,
       eingelassen_am: anmeldungTicketsTable.eingelassen_am,
     })
     .from(anmeldungTicketsTable)
     .innerJoin(thekeProfileTable,
       eq(thekeProfileTable.anmeldung_ticket_id, anmeldungTicketsTable.id))
     .where(and(
       isNotNull(anmeldungTicketsTable.eingelassen_am),
       eq(thekeProfileTable.tafel_ok, true),
     ))
     .orderBy(anmeldungTicketsTable.eingelassen_am);

Response:

    neuankoemmlinge,   // Array<{ id, person_name, scanned_at, lauter_song }>
    anwesende,         // Array<{ person_name, eingelassen_am }>
    server_time: new Date().toISOString(),
  });
