Problem: Ein Klick auf ein Porträt der Theke-Wand öffnet nichts. Ursache:
Der Container ruft in onPointerDown setPointerCapture() auf — dadurch wird das
click-Event vom Porträt weg auf den Container umgeleitet, der keinen Handler hat.
Fix: kein Pointer-Capture mehr; stattdessen Tap-vs-Drag über Bewegungsschwelle.
Das horizontale Ziehen der Wand MUSS weiter funktionieren.

    (e.currentTarget as HTMLDivElement).setPointerCapture(e.pointerId);

  const onPointerMove = useCallback((e: React.PointerEvent<HTMLDivElement>) => {
    if (!dragRef.current) return;

    const maxOffset  = Math.max(0, totalWidth - containerW + 40);
    setScrollOffset(Math.max(0, Math.min(maxOffset, dragRef.current.startOffset + delta)));
  }, [totalWidth]);

  const onPointerUp = useCallback(() => { dragRef.current = null; }, []);
