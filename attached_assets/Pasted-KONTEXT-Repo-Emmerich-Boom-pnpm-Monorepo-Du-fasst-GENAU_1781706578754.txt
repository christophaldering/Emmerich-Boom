UMSETZUNG PUNKT 1 — Foto-Resize
a) Ganz oben den Import ergänzen:

b) Direkt NACH der bestehenden Funktion storeFile diese Helper einfügen:

async function verkleinereBild(buffer: Buffer): Promise<{ buffer: Buffer; ext: string }> {
  try {
    const out = await sharp(buffer)
      .rotate()                                          // EXIF-Ausrichtung (Handyfotos!)
      .resize(1600, 1600, { fit: "inside", withoutEnlargement: true })
      .jpeg({ quality: 80, mozjpeg: true })
      .toBuffer();
    return { buffer: out, ext: "jpg" };
  } catch {
    return { buffer, ext: "jpg" };                       // Fallback: Original speichern
  }
}
